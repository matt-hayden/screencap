Uses a combination of the following FFMpeg commands:
[ '-an' ] 
[ '-skip_frame', 'nokey' ]
[ '-i', arg ]
[ '-f', 'image2' ]
[ '-vf', "select='eq(pict_type,PICT_TYPE_I)',tile" ]
[ '-vf', "select='eq(pict_type,PICT_TYPE_I)'" ]
[ '-vf', "select='eq(pict_type,I)'" ]
[ '-vf', "select='gt(scene\,0.9)'" ]
[ '-vf', "select='gt(scene\,0.4)',scale=160:120,tile" ]
[ '-vf', 'thumbnail=60' ]
[ '-frames:v', '60' ]
[ '-frames:v', '1' ]
[ '-vsync', '0' ]
[ '-vsync', 'vfr' ]
